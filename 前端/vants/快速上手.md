# 快速上手

## 安转包

~~~
pnpm add vant
~~~

## 安装组件自动导入插件

~~~
pnpm add unplugin-vue-components -D
~~~

## 在vite里配置插件

~~~
import vue from '@vitejs/plugin-vue';
import Components from 'unplugin-vue-components/vite';
import { VantResolver } from 'unplugin-vue-components/resolvers';

export default {
  plugins: [
    vue(),
    Components({
      resolvers: [VantResolver()],
    }),
  ],
};

~~~
