<script setup>  
/**  
 * TeacherDashboard.vue — 教师端主页  
 * 流程: 上传课件 → 解析 → 编辑脚本 → 预览播放  
 */  
import { ref } from 'vue'  
import Uploader from '../components/CourseGenerator/Uploader.vue'  
import ScriptEditor from '../components/CourseGenerator/ScriptEditor.vue'  
import AudioPlayer from '../components/CourseGenerator/AudioPlayer.vue'  
import { generateScript } from '../../api/aiService.js'  
  
const step = ref('upload') // upload | edit | preview  
const lessonId = ref('')  
const sections = ref([])  
const prompt = ref('')  
const audioUrl = ref('')  
const generating = ref(false)  
const genError = ref('')  
  
const handleMouseMove = (e) => {  
  if (!isResizing.value) return  
  // 限制最小和最大宽度  
  if (e.clientX > 150 && e.clientX < 500) {  
    sidebarWidth.value = e.clientX  
  }  
}  
  
const stopResizing = () => {  
  isResizing.value = false  
  document.removeEventListener('mousemove', handleMouseMove)  
  document.removeEventListener('mouseup', stopResizing)  
  document.body.style.cursor = 'default'  
}  
  
  
async function onParsed({ lessonId: id }) {  
  lessonId.value = id  
  generating.value = true  
  genError.value = ''  
  try {  
    const res = await generateScript(id)  
    sections.value = res?.script ?? res?.sections ?? []  
    prompt.value = res?.prompt ?? ''  
    step.value = 'edit'  
  } catch (e) {  
    genError.value = `脚本生成失败：${e.message || '未知错误'}`  
  } finally {  
    generating.value = false  
  }  
}  
  
function onSaved({ sections: s, prompt: p }) {  
  sections.value = s  
  prompt.value = p  
  // audioUrl 由后端 TTS 接口生成后返回，此处模拟占位  
  audioUrl.value = ''  
  step.value = 'preview'  
}  
</script>  
  
<template>  
  <div class="dashboard">  
    <!-- 侧栏 -->  
    <aside class="sidebar">  
      <div class="logo">  
        <span class="logo-text">超星智课</span>  
        <span class="logo-role">教师端</span>  
      </div>  
      <nav class="steps-nav">  
        <div :class="['step-item', { active: step === 'upload', done: step !== 'upload' }]">  
          <span class="step-dot">1</span><span>上传课件</span>  
        </div>  
        <div :class="['step-item', { active: step === 'edit' }]">  
          <span class="step-dot">2</span><span>编辑脚本</span>  
        </div>  
        <div :class="['step-item', { active: step === 'preview' }]">  
          <span class="step-dot">3</span><span>预览播放</span>  
        </div>  
      </nav>  
    </aside>  
  
    <div class="resizer" @mousedown="startResizing"></div>  
  
    <!-- 主内容区 -->  
    <main class="main-content">  
      <div class="center-container">  
        <section v-if="step === 'upload'" class="content-card">  
          <Uploader @parsed="onParsed" @error="genError = $event.message" />  
          <div v-if="generating" class="gen-hint">  
            <div class="spinner" /><span>AI 正在生成脚本，请稍候…</span>  
          </div>  
          <p v-if="genError" class="error-msg">{{ genError }}</p>  
        </section>  
  
        <section v-else-if="step === 'edit'" class="content-card">  
          <ScriptEditor  
              :lessonId="lessonId"  
              :initialSections="sections"  
              :initialPrompt="prompt"  
              @saved="onSaved"  
          />  
          <button class="btn-back" @click="step = 'upload'">← 重新上传</button>  
        </section>  
  
        <section v-else-if="step === 'preview'" class="content-card">  
          <h2 class="section-title">课程预览</h2>  
          <div v-if="!audioUrl" class="no-audio">  
            <p>TTS 音频生成中...</p>  
          </div>  
          <AudioPlayer v-else :audioUrl="audioUrl" :sections="sections" :lessonId="lessonId" />  
          <div class="preview-actions">  
            <button class="btn-back" @click="step = 'edit'">← 返回编辑</button>  
            <button class="btn-primary" @click="step = 'upload'">新建课程</button>  
          </div>  
        </section>  
      </div>  
    </main>  
  </div>  
</template>
