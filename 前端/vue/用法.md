# 用法

## v-model

将后面的数值转变成响应式变量，变量的初始值是挂载函数中的data(){这里的数值}

~~~js
const { createApp } = Vue;  
  
const appObj = {  
  data() {  
    return {  
      inputText: "",  
    };  
  },  
};  
  
createApp(appObj).mount("#app");
~~~

~~~html
<!DOCTYPE html>  
<html lang="en">  
<head>  
    <meta charset="UTF-8" />  
    <title>Vue Demo</title>  
	<meta name="viewport" content="width=device-width, initial-scale=1" />  
	<script src="https://unpkg.com/vue@3/dist/vue.global.js"></script>  
</head>  
<body>    
	<div id="app">  
	      <label for="input-demo">Vue Input:</label>  
	      <input id="input-demo" type="text" v-model="inputText" />  
	      <p>{{ inputText }}</p>  
    </div>  
    <hr />
    <!--只是一个普通的 HTML 标签，用来在页面上画一条水平分割线-->
    <div> 
	    <label for="input-demo">VanillaJS Input:</label>  
	    <input id="input-demo" class="input-text" type="text" />  
	    <p class="p-text"></p>  
    </div>  
    <script src="vanilla.js"></script>  
    <script src="vue.js"></script>  
  </body>  
</html>
~~~
## v-bind

绑定属性和vue参数一致


<script>  
  const { createApp } = Vue;  
  
  const app = {  
    data() {  
      return {  
        username: "",  
        is_username_error: "",  
  
        password: "",  
        is_password_error: "",  
      };  
    },  
  
    methods: {  
      handleSubmit() {  
        // event.preventDefault();  
  
        this.is_username_error = this.is_password_error = "";  
  
        if (this.username.length < 3 || this.password.length < 3) {  
          alert("username and password must be at least 3 characters");  
  
          if (this.username.length < 3) {  
            this.is_username_error = "input-error";  
          } else {  
            this.is_password_error = "input-error";  
          }  
  
          return;  
        }  
  
        alert(`username: ${this.username} \npassword: ${this.password}`);  
      },  
    },  
  };
~~~
